    """

    """

    if not os.path.exists(manifest_filename):

        return

    mp3_urls = []

    try:
        # Wrap the copy-pasted text block into a proper string tree for Python
        with open(manifest_filename, "r", encoding="utf-8") as f:
            xml_content = f.read().strip()
            
        # Parse the raw XML string directly
        root = ET.fromstring(xml_content)
        
        # Loop through every <file> tag block in the text dump
        for file_tag in root.findall('file'):
            original_tag = file_tag.find('original')
            
            # Extract the raw text path inside the <original></original> container
            if original_tag is not None and original_tag.text:
                raw_filename = original_tag.text.strip()
                
                # Verify it is actually an audio track asset
                if raw_filename.lower().endswith('.mp3'):
                    # Glue the pieces together using the clean global gateway and natural spaces
                    direct_link = "https://archive.org" + item_id + "/" + raw_filename
                    mp3_urls.append(direct_link)

    except ET.ParseError as pe:
        print(f"❌ XML Parsing Error: Make sure your manifest.txt starts with <files> and ends with </files>!")
        print(f"💡 Details: {pe}")
        return
    except Exception as e:
        print(f"❌ Error reading file block: {e}")
        return

    # Build the final flat database schema payload array
    payload = {
        "mp3s": mp3_urls
    }
    
    with open(output_filename, "w", encoding="utf-8") as f_out:
        json.dump(payload, f_out, indent=4)
        
    print(f"⚡ SUCCESS: {len(mp3_urls)} verified tracks extracted from the XML tree layout.")
    print(f"📦 Transmission data committed straight to: {output_filename}")
